LC701. 二叉搜索树中的插入操作

发表于2023-01-10|更新于2023-01-10|LeetCode二叉树专题
|阅读量:

解题思路

因为是二叉搜索树(Binary search tree),整个树是有序的，我们根据val来调整自己下一个要遍历的节点，若下一个节点为为空，则创立新节点，将节点插入即可.

C++

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func insertIntoBST(root *TreeNode, val int) *TreeNode {
    if root == nil {return &TreeNode{Val: val}}
    dfs(root, val)
    return root
}
func dfs(node *TreeNode, val int) {
    if node.Val > val && node.Left != nil {dfs(node.Left, val)}
    if node.Val < val && node.Right != nil {dfs(node.Right, val)}

    if node.Val > val && node.Left == nil {node.Left = &TreeNode{Val: val}}
    if node.Val < val && node.Right == nil {node.Right = &TreeNode{Val: val}}
}
